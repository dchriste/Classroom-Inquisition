01697         {

01699             FileStream fsInput = new FileStream(sInputFilename, FileMode.Open, FileAccess.Read);

01702 
01703             //actual crypto provider to be used on the file
01704             DESCryptoServiceProvider DES = new DESCryptoServiceProvider();
01705

01707             DES.Key = ASCIIEncoding.ASCII.GetBytes(sKey);

01709 
01710             //create a cryptostream class using the previous cryptographic provider and existing filestream obj
01711             ICryptoTransform desencrypt = DES.CreateEncryptor();
01712             CryptoStream cryptoctream = new CryptoStream(fsEncrypted, desencrypt, CryptoStreamMode.Write);
01713 
01714             //read in the input and write the encypted output file by passing it to cryptostream
01715             byte[] bytearrayinput = new byte[fsInput.Length];
