01408         static void EncryptFile(string sInputFilename, string sOutputFilename, string sKey)
01409         {
01410             //input filestream
01411             FileStream fsInput = new FileStream(sInputFilename, FileMode.Open, FileAccess.Read);
01412             //output filestream
01413             FileStream fsEncrypted = new FileStream(sOutputFilename, FileMode.Create, FileAccess.Write);
01414 
01415             //actual crypto provider to be used on the file
01416             DESCryptoServiceProvider DES = new DESCryptoServiceProvider();
01417 
01418             //give the cryptographic provider the secret key and init vector otherwise it will make them up...
01419             DES.Key = ASCIIEncoding.ASCII.GetBytes(sKey);
01420             DES.IV = ASCIIEncoding.ASCII.GetBytes(sKey);

01422             //create a cryptostream class using the previous cryptographic provider and existing filestream obj

01424             CryptoStream cryptoctream = new CryptoStream(fsEncrypted, desencrypt, CryptoStreamMode.Write);

01428             fsInput.Read(bytearrayinput, 0, bytearrayinput.Length);
