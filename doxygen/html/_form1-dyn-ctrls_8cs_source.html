02381         static void EncryptFile(string sInputFilename, string sOutputFilename, string sKey)
02382         {
02383             //input filestream
02384             FileStream fsInput = new FileStream(sInputFilename, FileMode.Open, FileAccess.Read);
02385             //output filestream
02386             FileStream fsEncrypted = new FileStream(sOutputFilename, FileMode.Create, FileAccess.Write);
02387 
02388             //actual crypto provider to be used on the file
02389             DESCryptoServiceProvider DES = new DESCryptoServiceProvider();
02390 
02391             //give the cryptographic provider the secret key and init vector otherwise it will make them up...
02392             DES.Key = ASCIIEncoding.ASCII.GetBytes(sKey);
02393             DES.IV = ASCIIEncoding.ASCII.GetBytes(sKey);
02394 
02395             //create a cryptostream class using the previous cryptographic provider and existing filestream obj
02396             ICryptoTransform desencrypt = DES.CreateEncryptor();
02397             CryptoStream cryptoctream = new CryptoStream(fsEncrypted, desencrypt, CryptoStreamMode.Write);
02398 
02399             //read in the input and write the encypted output file by passing it to cryptostream
02400             byte[] bytearrayinput = new byte[fsInput.Length];
02401             fsInput.Read(bytearrayinput, 0, bytearrayinput.Length);

02407         {
02408             DESCryptoServiceProvider DES = new DESCryptoServiceProvider();
02409             DES.Key = ASCIIEncoding.ASCII.GetBytes(sKey);
02410             DES.IV = ASCIIEncoding.ASCII.GetBytes(sKey);
02411 
02412             //create file stream to read the encrypted file back
02413             FileStream fsread = new FileStream(sInputFilename, FileMode.Open, FileAccess.Read);
02414 
02415             //create aes decryptor from the aes instance
02416             ICryptoTransform desdecrypt = DES.CreateDecryptor();
02417 
02418             //create crypto stream set to read and do a aes decryption transform on incoming bytes
02419             CryptoStream cryptostreamDecr = new CryptoStream(fsread, desdecrypt, CryptoStreamMode.Read);
02420

02422             StreamWriter fsDecrypted = new StreamWriter(sOutputFilename);
02423             fsDecrypted.Write(new StreamReader(cryptostreamDecr).ReadToEnd());
02424             fsDecrypted.Flush();
